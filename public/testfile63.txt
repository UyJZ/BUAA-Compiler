int FourWhile() {
  int a;
  a = 5;
  int b;
  int c;
  b = 6;
  c = 7;
  int d;
  d = 10;
  for (;a < 20;) {
    a = a + 3;
    for (;b < 10;){
      b = b + 1;
      for (;c == 7;){
        c = c - 1;
        for (;d < 20;){
          d = d + 3;
        }
        d = d - 1;
      }
      c = c + 1;
    }
    b = b - 2;
  }
  
  return (a + (b + d) + c);
}
